local espBtn = createButton("ESP: OFF", UDim2.new(0.05, 0, 0.34, 0))
local highlights = {}
local espColors = {
    enemy = Color3.new(1, 0, 0),    -- –ö—Ä–∞—Å–Ω—ã–π –¥–ª—è –≤—Ä–∞–≥–æ–≤
    teammate = Color3.new(0, 0, 1), -- –°–∏–Ω–∏–π –¥–ª—è —Ç–∏–º–º–µ–π—Ç–æ–≤
    neutral = Color3.new(0, 1, 0)   -- –ó–µ–ª–µ–Ω—ã–π –¥–ª—è –Ω–µ–π—Ç—Ä–∞–ª—å–Ω—ã—Ö

espBtn.MouseButton1Click:Connect(function()
    espEnabled = not espEnabled
    
    if espEnabled then
        espBtn.Text = "ESP: ON"
        espBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 0)
        print("‚úÖ ESP ON")
        
        -- –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è —Ü–≤–µ—Ç–∞ ESP
        local function getEspColor(targetPlayer)
            if player.Team and targetPlayer.Team then
                if player.Team == targetPlayer.Team then
                    return espColors.teammate
                else
                    return espColors.enemy
                end
            end
            return espColors.neutral
        end
        
        -- –í–∫–ª—é—á–∏—Ç—å ESP –¥–ª—è –≤—Å–µ—Ö –∏–≥—Ä–æ–∫–æ–≤
        for _, otherPlayer in pairs(Players:GetPlayers()) do
            if otherPlayer ~= player and otherPlayer.Character then
                local highlight = Instance.new("Highlight")
                highlight.FillColor = getEspColor(otherPlayer)
                highlight.OutlineColor = Color3.new(1, 1, 1)
                highlight.FillTransparency = 0.5
                highlight.OutlineTransparency = 0
                highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                highlight.Parent = otherPlayer.Character
                highlight.Adornee = otherPlayer.Character
                highlights[otherPlayer] = highlight
                
                print("üî¥ ESP added for: " .. otherPlayer.Name)
            end
        end
        
        -- –û–±—Ä–∞–±–æ—Ç–∫–∞ –Ω–æ–≤—ã—Ö –∏–≥—Ä–æ–∫–æ–≤
        Players.PlayerAdded:Connect(function(newPlayer)
            if espEnabled then
                newPlayer.CharacterAdded:Connect(function(character)
                    wait(1)
                    if espEnabled and character then
                        local highlight = Instance.new("Highlight")
                        highlight.FillColor = getEspColor(newPlayer)
                        highlight.OutlineColor = Color3.new(1, 1, 1)
                        highlight.FillTransparency = 0.5
                        highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                        highlight.Parent = character
                        highlight.Adornee = character
                        highlights[newPlayer] = highlight
                        print("üî¥ ESP added for new player: " .. newPlayer.Name)
                    end
                end)
            end
        end)
        
    else
        espBtn.Text = "ESP: OFF"
        espBtn.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
        print("‚ùå ESP OFF")
        
        -- –í—ã–∫–ª—é—á–∏—Ç—å ESP
        for _, highlight in pairs(highlights) do
            if highlight then
                highlight:Destroy()
            end
        end
        highlights = {}
    end
end)

local wallshootBtn = createButton("WALLSHOOT: OFF", UDim2.new(0.05, 0, 0.52, 0))
wallshootBtn.MouseButton1Click:Connect(function()
    wallshootEnabled = not wallshootEnabled
    
    if wallshootEnabled then
        wallshootBtn.Text = "WALLSHOOT: ON"
        wallshootBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 0)
        print("‚úÖ Wallshoot ON - Shooting through walls enabled")
        
        -- –§—É–Ω–∫—Ü–∏—è –¥–ª—è —Å—Ç—Ä–µ–ª—å–±—ã —á–µ—Ä–µ–∑ —Å—Ç–µ–Ω—ã
        local function setupWallshoot()
            -- –ü–æ–ª—É—á–∞–µ–º –æ—Ä—É–∂–∏–µ –∏–≥—Ä–æ–∫–∞
            local character = player.Character
            if not character then return end
            
            -- –ò—â–µ–º –∏–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç—ã/–æ—Ä—É–∂–∏–µ
            for _, tool in pairs(character:GetChildren()) do
                if tool:IsA("Tool") then
                    -- –ü–µ—Ä–µ—Ö–≤–∞—Ç—ã–≤–∞–µ–º —Å–æ–±—ã—Ç–∏—è —Å—Ç—Ä–µ–ª—å–±—ã
                    local toolScript = tool:FindFirstChildOfClass("LocalScript")
                    if toolScript then
                        -- –ó–¥–µ—Å—å –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å –ª–æ–≥–∏–∫–∞ –º–æ–¥–∏—Ñ–∏–∫–∞—Ü–∏–∏ –æ—Ä—É–∂–∏—è –¥–ª—è —Å—Ç—Ä–µ–ª—å–±—ã —á–µ—Ä–µ–∑ —Å—Ç–µ–Ω—ã
                        -- –≠—Ç–æ –∑–∞–≤–∏—Å–∏—Ç –æ—Ç –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–π –∏–≥—Ä—ã
                        print("üî´ Tool found: " .. tool.Name)
                    end
                end
            end
            
            -- –ê–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤–Ω—ã–π –º–µ—Ç–æ–¥: –∏–∑–º–µ–Ω—è–µ–º Raycast –ø–∞—Ä–∞–º–µ—Ç—Ä—ã
            local function modifyRaycasts()
                -- –≠—Ç–æ—Ç –º–µ—Ç–æ–¥ —Ä–∞–±–æ—Ç–∞–µ—Ç –¥–ª—è –∏–≥—Ä, –∏—Å–ø–æ–ª—å–∑—É—é—â–∏—Ö Raycast –¥–ª—è –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –ø–æ–ø–∞–¥–∞–Ω–∏–π
                local oldNamecall
                oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
                    local method = getnamecallmethod()
                    local args = {...}
                    
                    if wallshootEnabled and method == "FindPartOnRayWithIgnoreList" then
                        -- –ò–∑–º–µ–Ω—è–µ–º Raycast —á—Ç–æ–±—ã –∏–≥–Ω–æ—Ä–∏—Ä–æ–≤–∞—Ç—å –≤—Å–µ —Å—Ç–µ–Ω—ã
                        local originalIgnoreList = args[2] or {}
                        local newIgnoreList = {}
                        
                        -- –î–æ–±–∞–≤–ª—è–µ–º –≤—Å–µ —Å—Ç–µ–Ω—ã –≤ –∏–≥–Ω–æ—Ä –ª–∏—Å—Ç
                        for _, part in pairs(Workspace:GetDescendants()) do
                            if part:IsA("BasePart") and part.Name:lower():find("wall") or part.Name:lower():find("part") then
                                table.insert(newIgnoreList, part)
                            end
                        end
                        
                        -- –û–±—ä–µ–¥–∏–Ω—è–µ–º —Å–ø–∏—Å–∫–∏
                        for _, item in pairs(originalIgnoreList) do
                            table.insert(newIgnoreList, item)
                        end
                        
                        args[2] = newIgnoreList
                        return oldNamecall(self, unpack(args))
                    end
                    
                    return oldNamecall(self, ...)
                end)
            end
            
            -- –ü—ã—Ç–∞–µ–º—Å—è –º–æ–¥–∏—Ñ–∏—Ü–∏—Ä–æ–≤–∞—Ç—å Raycast —Å–∏—Å—Ç–µ–º—É
            pcall(modifyRaycasts)
        end
        
        setupWallshoot()
        
    else
        wallshootBtn.Text = "WALLSHOOT: OFF"
        wallshootBtn.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
        print("‚ùå Wallshoot OFF")
    end
end)

local aimbotBtn = createButton("AIMBOT: OFF", UDim2.new(0.05, 0, 0.58, 0))
local aimbotConnection
local fovCircle = nil
local aimbotFov = 100  -- FOV –¥–ª—è –∞–∏–º–±–æ—Ç–∞

-- –°–æ–∑–¥–∞–µ–º FOV –∫—Ä—É–≥ –¥–ª—è –∞–∏–º–±–æ—Ç–∞
local function createFovCircle()
    if fovCircle then fovCircle:Remove() end
    
    local circle = Drawing.new("Circle")
    circle.Visible = true
    circle.Thickness = 2
    circle.Color = Color3.new(1, 1, 1)
    circle.Transparency = 1
    circle.Filled = false
    circle.Radius = aimbotFov
    circle.Position = Vector2.new(
        workspace.CurrentCamera.ViewportSize.X / 2,
        workspace.CurrentCamera.ViewportSize.Y / 2
    )
    
    fovCircle = circle
    return circle
end

-- –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –ø–æ–∑–∏—Ü–∏–∏ FOV –∫—Ä—É–≥–∞
RunService.RenderStepped:Connect(function()
    if fovCircle then
        fovCircle.Position = Vector2.new(
            workspace.CurrentCamera.ViewportSize.X / 2,
            workspace.CurrentCamera.ViewportSize.Y / 2
        )
    end
end)

aimbotBtn.MouseButton1Click:Connect(function()
    aimbotEnabled = not aimbotEnabled
    
    if aimbotEnabled then
        aimbotBtn.Text = "AIMBOT: ON"
        aimbotBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 0)
        print("‚úÖ Aimbot ON - Hold Right Mouse to aim")
        
        -- –°–æ–∑–¥–∞–µ–º FOV –∫—Ä—É–≥
        createFovCircle()
        fovCircle.Visible = true
        
        -- –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–∏—Å–∫–∞ –≤—Ä–∞–≥–∞ –≤ FOV
        local function getClosestEnemyInFov()
            local closestPlayer = nil
            local closestDistance = math.huge
            local localRoot = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
            local camera = workspace.CurrentCamera
            
            if not localRoot or not camera then return nil end
            
            local mousePos = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y / 2)
            
            for _, otherPlayer in pairs(Players:GetPlayers()) do
                if otherPlayer ~= player and otherPlayer.Character then
                    -- –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ —Ç–∏–º–º–µ–π—Ç–æ–≤
                    local isTeammate = false
                    
                    if player.Team and otherPlayer.Team then
                        if player.Team == otherPlayer.Team then
                            isTeammate = true
                        end
                    end
                    
                    if not isTeammate then
                        local targetRoot = otherPlayer.Character:FindFirstChild("HumanoidRootPart")
                        local humanoid = otherPlayer.Character:FindFirstChildOfClass("Humanoid")
                        
                        if targetRoot and humanoid and humanoid.Health > 0 then
                            -- –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä—É–µ–º –º–∏—Ä–æ–≤—É—é –ø–æ–∑–∏—Ü–∏—é –≤ —ç–∫—Ä–∞–Ω–Ω—É—é
                            local screenPoint, onScreen = camera:WorldToViewportPoint(targetRoot.Position)
                            
                            if onScreen then
                                local screenPos = Vector2.new(screenPoint.X, screenPoint.Y)
                                local distance = (mousePos - screenPos).Magnitude
                                
                                -- –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –ª–∏ –≤—Ä–∞–≥ –≤ FOV
                                if distance <= aimbotFov and distance < closestDistance then
                                    closestDistance = distance
                                    closestPlayer = otherPlayer
                                end
                            end
                        end
                    end
                end
            end
            
            return closestPlayer
        end
        
        -- –í–∫–ª—é—á–∏—Ç—å –∞–∏–º–±–æ—Ç
        aimbotConnection = RunService.Heartbeat:Connect(function()
            if aimbotEnabled and UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then
                local closestEnemy = getClosestEnemyInFov()
                if closestEnemy and closestEnemy.Character then
                    local targetRoot = closestEnemy.Character:FindFirstChild("HumanoidRootPart")
                    local camera = workspace.CurrentCamera
                    if targetRoot and camera then
                        -- –ü–ª–∞–≤–Ω—ã–π –∞–∏–º–±–æ—Ç
                        local currentCFrame = camera.CFrame
                        local targetCFrame = CFrame.new(camera.CFrame.Position, targetRoot.Position)
                        camera.CFrame = currentCFrame:Lerp(targetCFrame, 0.7)
                    end
                end
            end
        end)
        
    else
        aimbotBtn.Text = "AIMBOT: OFF"
        aimbotBtn.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
        print("‚ùå Aimbot OFF")
        
        -- –í—ã–∫–ª—é—á–∏—Ç—å –∞–∏–º–±–æ—Ç –∏ —Å–∫—Ä—ã—Ç—å FOV –∫—Ä—É–≥
        if aimbotConnection then
            aimbotConnection:Disconnect()
        end
        if fovCircle then
            fovCircle.Visible = false
        end
    end
end)
